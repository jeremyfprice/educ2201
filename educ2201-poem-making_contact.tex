\poemtitle{Making Contact}
%\settowidth{\versewidth}{I can conceive of having}
\begin{verse}%[\versewidth]
I believe\\
The greatest gift\\
I can conceive of having\\
from anyone\\
is\\
to be seen by them,\\
heard by them,\\
to be understood\\
and\\
touched by them.\\
The greatest gift\\
I can give\\
is\\
to see, hear, understand\\
and to touch\\
another person.\\
When this is done\\
I feel\\
contact has been made.\\
\end{verse}
\attrib{Virginia Satir}
